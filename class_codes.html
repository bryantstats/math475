Auto-regressive Models

library(ggfortify)
Warning: package 'ggfortify' was built under R version 4.4.2
Loading required package: ggplot2
set.seed(2023)
n = 100000
y = rep(0, n)
y[1] = 0
b0 = 0
b1 = 1.0001
e = rnorm(n, sd = 1)
for (t in 2:n)
{
  y[t] = b0 + b1*y[t-1]+e[t]
}
autoplot(ts(y)) + ggtitle(paste0("An Autoregressive series with beta1 = ",b1))
